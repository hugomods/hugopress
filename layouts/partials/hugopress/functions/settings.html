".")) }} {{- end }} {{- end }} {{- $hooks.Add $name (slice $option) }} {{- end }} {{- end }} {{- with .Attributes }} {{- range $name, $option := . }} {{- if or $option.disable $option.disabled }} {{- if site.Params.hugopress.debug }} {{- warnf "[hugopress] [%s] attribute %q was disabled." $moduleName $name }} {{- end }} {{- continue }} {{- end }} {{- $partial := printf "hugopress/modules/%s/attributes/%s" $moduleName $name }} {{- if not (partialCached "hugopress/functions/partial-exists" $partial $partial) }} {{- warnf "[hugopress] [%s] attribute does't exist: layouts/partials/%s.html" $moduleName $partial }} {{- continue }} {{- end }} {{- $option = merge $option (dict "name" $name "module" $moduleName "partial" $partial) }} {{- $attributes.Add $name (slice $option) }} {{- end }} {{- end }} {{- end }} {{- return dict "hooks" $hooks.Values "attributes" $attributes.Values }}
